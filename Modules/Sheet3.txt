Public Sub InitShopLoad()
    Dim fso As FileSystemObject
    Set fso = New FileSystemObject
    Dim shopQuery As String
    shopQuery = fso.OpenTextFile(Config.QUERY_PATH & "JobLoad.sql", ForReading).ReadAll()
    
    Dim results() As Variant
    ThisWorkbook.SQLQuery shopQuery, Connections.E10
    results = ThisWorkbook.ResultRecordSet.GetRows()
    results = Application.Transpose(results)
    
    Dim testRange As Range
    Set testRange = Range(Cells(2, 1), Cells(UBound(results, 1) + 1, UBound(results, 2)))
    testRange = results
    Debug.Print ("stop here")

End Sub
